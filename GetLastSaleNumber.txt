[HttpGet("get-latest-sale-number")]
public async Task<IActionResult> GetLatestSaleNumber()
{
    var lastSale = await _context.Sales
        .OrderByDescending(s => s.SaleId)
        .FirstOrDefaultAsync();

    if (lastSale == null)
        return Ok(new { nextNumber = 1 });

    // Last sale number format: Sale_18_9_2025_7
    var parts = lastSale.SaleNumber.Split('_');
    var lastCounter = int.Parse(parts[^1]); // last part
    return Ok(new { nextNumber = lastCounter + 1 });
}

[HttpPost("create-sale")]
public async Task<IActionResult> CreateSale([FromBody] Sale sale)
{
    var lastSale = await _context.Sales
        .OrderByDescending(s => s.SaleId)
        .FirstOrDefaultAsync();

    var nextCounter = 1;
    if (lastSale != null)
    {
        var parts = lastSale.SaleNumber.Split('_');
        nextCounter = int.Parse(parts[^1]) + 1;
    }

    var date = DateTime.Now;
    sale.SaleNumber = $"Sale_{date.Day}_{date.Month}_{date.Year}_{nextCounter}";
    sale.SaleDate = date;

    _context.Sales.Add(sale);
    await _context.SaveChangesAsync();

    return Ok(sale);
}
