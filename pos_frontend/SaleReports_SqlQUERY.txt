--Customer Sale History Query:
select s.CustomerId, SUM(si.Quantity) as TotalQuantity, Sum(si.TotalPrice) as TotalRenevue 
  from [POS_System].[dbo].[SaleItems] si join [POS_System].[dbo].[Sales] s on si.SaleId = s.SaleId where si.TotalPrice > 0
  group by s.CustomerId order by TotalRenevue DESC;

--Product Wise Sale Histry Query:
select p.ProductId, p.ProductName, SUM(si.Quantity) as TotalQuantity, SUM(si.TotalPrice) as TotalRenevue  
  from [POS_System].[dbo].[SaleItems] si join [POS_System].[dbo].[Products] p on si.ProductId = p.ProductId 
  group by p.ProductId, p.ProductName order by TotalRenevue Desc;

--Daily Sales Report Query:
select CAST(s.SaleDate as date) as SaleDate, Sum(si.TotalPrice) as TotalRenevue 
  from [POS_System].[dbo].[SaleItems] si join [POS_System].[dbo].[Sales] s on si.SaleId = s.SaleId
  group by CAST(s.SaleDate as date) order by TotalRenevue DESC;

--Profit Report Query:
SELECT 
    p.ProductId, p.ProductName, p.CostPrice, p.SellingPrice, (p.SellingPrice - p.CostPrice) AS ProfitMargin
FROM [POS_System].[dbo].[Products] p;

--Customer filter using Date Range (On Report Page):

join [POS_System].[dbo].[Customers] c on s.CustomerId = c.CustomerId join [POS_System].[dbo].[Products] p
on si.ProductId = p.ProductId where c.CustomerName = 'Hammad' and CAST(s.SaleDate AS date) BETWEEN '2025-09-04' and '2025-09-09';

--Customer Aging Report:
Declare @ReportDate Date = '2025-10-10';
  select c.CustomerName, 
  SUM(Case when datediff(DAY, s.SaleDate, @ReportDate) <= 30 then (si.TotalPrice - si.AmountPay) else 0 end) as [0_30_Days], 
  SUM(Case when datediff(DAY, s.SaleDate, @ReportDate) between 31 And 60 then (si.TotalPrice - si.AmountPay) else 0 end) as [31_60_Days]
  from [POS_System].[dbo].[SaleItems] si join [POS_System].[dbo].[Sales] s on si.SaleId = s.SaleId join 
  [POS_System].[dbo].[Customers] c on s.CustomerId = c.CustomerId;
